# vimrc
vim setting

# Install
1. Install vim
 ```
 $sudo apt update
 $sudo apt install vim
 $sudo apt install vim-gnome
 ```

1. Install NeoBundle
 ```
 $ mkdir -p ~/.vim/bundle
 $ git clone git://github.com/Shougo/neobundle.vim ~/.vim/bundle/neobundle.vim
 ```
2. Download vimrc
 ```
 $ cd ~/
 $ git clone https://github.com/naoki-sh/vimrc.git
 $ cp ~/vimrc/.vimrc ~/
 $ sudo rm -r ~/vimrc
 ```

3. run and install vim modules
```
 $ vim
```
